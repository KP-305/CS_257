1. Rocket Images
SELECT ?rocket ?rocketLabel ?image
WHERE {
  ?rocket wdt:P31/wdt:P279* wd:Q728.
  ?rocket wdt:P18 ?image.
  SERVICE wikibase:label { bd:serviceParam wikibase:language "[AUTO_LANGUAGE],en". }
}
LIMIT 10

2. Items from the Year 1985
SELECT ?item ?itemLabel ?date
WHERE {
  ?item wdt:P571 ?date.
  FILTER(YEAR(?date) = 1985)
  SERVICE wikibase:label { bd:serviceParam wikibase:language "[AUTO_LANGUAGE],en". }
}
LIMIT 10
